from flask import Flask, render_template, request, redirect import pandas as pd from datetime import datetime import os app = Flask(__name__) # File paths TOWER_FILE = os.path.join("data", "tower1.xlsx") LOG_FOLDER = os.path.join("logs") # Ensure logs folder exists os.makedirs(LOG_FOLDER, exist_ok=True) @app.route("/", methods=["GET"]) def index(): try: df = pd.read_excel(TOWER_FILE, sheet_name="ClientList") client_ids = df["Client ID"].dropna().unique() return render_template("client_links.html", client_ids=client_ids) except Exception as e: return f"Error loading client list: {e}", 500 @app.route("/client/", methods=["GET"]) def client_view(client_id): try: df = pd.read_excel(TOWER_FILE, sheet_name=client_id) if "Action Item" not in df.columns or "Status" not in df.columns: return "Invalid sheet format for client.", 400 df = df[["Action Item", "Status"]].fillna("") items = list(df.iterrows()) status_options = ["To Do", "In Progress", "Complete", "Backlog"] return render_template( "client_view.html", client_id=client_id, items=items, status_options=status_options ) except Exception as e: return f"Error loading client view: {e}", 500 @app.route("/submit", methods=["POST"]) def submit(): client_id = request.form.get("client_id") if not client_id: return "Client ID is missing.", 400 try: df = pd.read_excel(TOWER_FILE, sheet_name=client_id) if "Action Item" not in df.columns: return "Invalid sheet format for client.", 400 timestamp = datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S") log_rows = [] for idx, row in df.iterrows(): action = row.get("Action Items", "") new_status = request.form.get(f"status_{idx}") if action and new_status: log_rows.append([client_id, timestamp, action, new_status]) df.at[idx, "Status"] = new_status # Save log if log_rows: log_df = pd.DataFrame(log_rows, columns=["Client ID", "Timestamp", "Action Items", "Status"]) log_path = os.path.join(LOG_FOLDER, f"log_{client_id}.csv") if os.path.exists(log_path): existing_log = pd.read_csv(log_path) log_df = pd.concat([existing_log, log_df], ignore_index=True) log_df.to_csv(log_path, index=False) # Save updated status back to the Excel file with pd.ExcelWriter(TOWER_FILE, engine='openpyxl', mode='a', if_sheet_exists='replace') as writer: df.to_excel(writer, sheet_name=client_id, index=False) return redirect("/") except Exception as e: return f"Error submitting updates: {e}", 500 # Required for Render deployment: bind to host and port from environment if __name__ == "__main__": port = int(os.environ.get("PORT", 5000)) app.run(host="0.0.0.0", port=port, debug=True)
